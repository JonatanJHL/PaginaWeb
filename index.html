Ejemplos de uso de clases y funciones avanzadas en JavaScript:

Ejemplo de Clases


class Usuario {
    constructor(nombre, rol) {
        this.nombre = nombre;
        this.rol = rol;
    }

    saludar() {
        console.log(`Hola, soy ${this.nombre} y mi rol es ${this.rol}`);
    }
}

const usuario1 = new Usuario('Jonatan', 'Desarrollador');
usuario1.saludar();  // Hola, soy Jonatan y mi rol es Desarrollador

Funciones Asíncronas


async function obtenerDatos() {
    try {
        let respuesta = await fetch('https://jsonplaceholder.typicode.com/posts');
        let datos = await respuesta.json();
        console.log(datos);
    } catch (error) {
        console.error('Error al obtener datos:', error);
    }
}

obtenerDatos();
