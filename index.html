Consulta SQL para Reportes NOC 
WITH log_tiempos AS (
    SELECT 
        ticket_id,
        MIN(CASE WHEN accion LIKE '%Nuevo%' THEN fecha END) AS fecha_inicio,
        MAX(CASE WHEN accion LIKE '%Resuelto%' THEN fecha END) AS fecha_fin
    FROM sistema.logs_tickets
    GROUP BY ticket_id
)
SELECT
    t.id AS ID_Ticket,
    c.nombre_empresa AS Cliente,
    t.prioridad,
    t.estado,
    TIMESTAMPDIFF(SECOND, lt.fecha_inicio, lt.fecha_fin) AS tiempo_respuesta_segundos
FROM 
    sistema.tickets t
JOIN sistema.clientes c ON t.cliente_id = c.id
LEFT JOIN log_tiempos lt ON lt.ticket_id = t.id;
