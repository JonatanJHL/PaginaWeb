Consulta SQL para Reportes NOC 
WITH log_tiempos AS (
                            SELECT 
                                ticket_id,
                                MIN(CASE WHEN accion LIKE '%Nuevo%' THEN fecha END) AS fecha_inicio,
                                MAX(CASE WHEN accion LIKE '%Resuelto%' THEN fecha END) AS fecha_fin
                            FROM sistema.logs_tickets
                            GROUP BY ticket_id
                        )
                        SELECT
                            t.id AS ID_Ticket,
                            c.nombre_empresa AS Cliente,
                            t.prioridad,
                            t.estado,
                            TIMESTAMPDIFF(SECOND, lt.fecha_inicio, lt.fecha_fin) AS tiempo_respuesta_segundos
                        FROM 
                            sistema.tickets t
                        JOIN sistema.clientes c ON t.cliente_id = c.id
                        LEFT JOIN log_tiempos lt ON lt.ticket_id = t.id;

Detección Facial con OpenCV (Python) 
import cv2
                        import numpy as np

                        # Cargar modelo preentrenado
                        face_cascade = cv2.CascadeClassifier('haarcascade_frontalface_default.xml')
                        emotion_model = load_emotion_model()

                        def detect_emotions(frame):
                            gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
                            faces = face_cascade.detectMultiScale(gray, 1.3, 5)
                            
                            for (x,y,w,h) in faces:
                                roi = gray[y:y+h, x:x+w]
                                roi = cv2.resize(roi, (48,48))
                                emotion = emotion_model.predict(roi[np.newaxis,:,:,np.newaxis])
                                
                                # Dibujar rectángulo y etiqueta
                                cv2.rectangle(frame,(x,y),(x+w,y+h),(255,0,0),2)
                                cv2.putText(frame, emotion, (x,y), cv2.FONT_HERSHEY_SIMPLEX, 1, (0,255,0), 2)
                        
                    return frame
